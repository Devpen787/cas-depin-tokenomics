% Heading-space guard without external packages: avoid headings near page bottoms.
\newcommand{\thesisneedspace}[1]{%
  \par
  \begingroup
  \ifdim\pagegoal<\maxdimen
    \dimen0=\pagegoal
    \advance\dimen0 by -\pagetotal
    \ifdim\dimen0<#1\newpage\fi
  \fi
  \endgroup
}
\preto\section{\thesisneedspace{6\baselineskip}}
